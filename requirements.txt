# src_next 有声书 WebUI 运行依赖
#
# Python 版本要求：>= 3.10（代码使用 PEP 604 的 `str | None` 语法）
# 推荐版本：Python 3.11 或 3.12
#
# 创建 conda 环境：
#   conda create -n audiobook python=3.11 -y
#   conda activate audiobook
#   pip install -r requirements.txt
#
# 说明：
#   * 本 requirements 只覆盖 src_next/ WebUI 主机端运行所需；
#   * 模型服务（LLM / Voicebank / TTS）跑在各自的服务器上，
#     通过 HTTP API 调用，因此本机**不需要**装 torch / transformers /
#     accelerate / librosa 等大型 ML 库；
#   * 以下版本下限是经过验证可用的最低版本；新版本通常也兼容。

# ─── WebUI 框架 ────────────────────────────────────────────────────
# Gradio 4.x 推荐；代码兼容 3.x（自动 fallback queue API）
gradio>=4.0.0

# ─── HTTP 客户端 ───────────────────────────────────────────────────
# 调用 Gemma4 / Qwen3 / CosyVoice3 / IndexTTS 等 HTTP API
requests>=2.28.0
urllib3>=1.26.0

# ─── 配置加载 ──────────────────────────────────────────────────────
# yaml profile 加载（yaml.safe_load）
PyYAML>=6.0

# ─── 音频处理（仅 cosyvoice_http adapter 需要） ────────────────────
# 把 IEEE float wav 转 PCM_16 落盘
numpy>=1.23.0
soundfile>=0.12.0
